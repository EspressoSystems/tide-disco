12/// A function to add error information to a response body.
13///
14/// This trait is object safe, so it can be used to dynamically dispatch to different strategies for
15/// serializing the error depending on the format version being used.

25/// Type-erase a format-specific error handler.
26pub(crate) fn error_handler<Error, VER>() -> Arc<dyn ErrorHandler<Error>>
27where
28    Error: crate::Error,
29    VER: StaticVersionType,
30{
31    Arc::new(|accept, error, res| {
32        // Try to add the error to the response body using a format accepted by the client. If we
33        // cannot do that (for example, if the client requested a format that is incompatible with a
34        // serialized error) just add the error as a string using plaintext.
35        let (body, content_type) = route::response_body::<_, Error, VER>(accept, &error)
36            .unwrap_or_else(|_| (error.to_string().into(), mime::PLAIN));
37        res.set_body(body);
38        res.set_content_type(content_type);
39    })
40}

42/// Server middleware which automatically populates the body of error responses.
43///
44/// If the response contains an error, the error is encoded into the [Error](crate::Error) type
45/// (either by downcasting if the server has generated an instance of [Error](crate::Error), or by
46/// converting to a [String] using [Display] if the error can not be downcasted to
47/// [Error](crate::Error)). The resulting [Error](crate::Error) is then serialized and used as the
48/// body of the response.
49///
50/// If the response does not contain an error, it is passed through unchanged.
51pub(crate) struct AddErrorBody<E> {
52    handler: Arc<dyn ErrorHandler<E>>,
53}
54
55impl<E> AddErrorBody<E> {
56    pub(crate) fn new(handler: Arc<dyn ErrorHandler<E>>) -> Self {
57        Self { handler }
58    }
59
60    pub(crate) fn with_version<VER>() -> Self
61    where
62        E: crate::Error,
63        VER: StaticVersionType,
64    {
65        Self::new(error_handler::<E, VER>())
66    }
67}

69impl<T, E> tide::Middleware<T> for AddErrorBody<E>
70where
71    T: Clone + Send + Sync + 'static,
72    E: crate::Error,
73{
74    fn handle<'a, 'b, 't>(
75        &'a self,
76        req: tide::Request<T>,
77        next: tide::Next<'b, T>,
78    ) -> BoxFuture<'t, tide::Result>
79    where
80        'a: 't,
81        'b: 't,
82        Self: 't,
83    {
84        async {
85            let accept = RequestParams::accept_from_headers(&req)?;
86            let mut res = next.run(req).await;
87            if let Some(error) = res.take_error() {
88                let error = E::from_server_error(error);
89                tracing::info!("responding with error: {}", error);
90                (self.handler)(&accept, &error, &mut res);
91                Ok(res)
92            } else {
93                Ok(res)
94            }
95        }
96        .boxed()
97    }
98}

116impl<State, Error> tide::Middleware<Arc<App<State, Error>>> for MetricsMiddleware
117where
118    State: Send + Sync + 'static,
119    Error: crate::Error + Send + Sync + 'static,
120{
121    fn handle<'a, 'b, 't>(
122        &'a self,
123        req: tide::Request<Arc<App<State, Error>>>,
124        next: tide::Next<'b, Arc<App<State, Error>>>,
125    ) -> BoxFuture<'t, tide::Result>
126    where
127        'a: 't,
128        'b: 't,
129        Self: 't,
130    {
131        let route = self.route.clone();
132        let api = self.api.clone();
133        let version = self.api_version;
134        async move {
135            if req.method() != http::Method::Get {
136                // Metrics only apply to GET requests. For other requests, proceed with normal
137                // dispatching.
138                return Ok(next.run(req).await);
139            }
140            // Look at the `Accept` header. If the requested content type is plaintext, we consider
141            // it a metrics request. Other endpoints have typed responses yielding either JSON or
142            // binary.
143            let accept = RequestParams::accept_from_headers(&req)?;
144            let reponse_ty =
145                best_response_type(&accept, &[mime::PLAIN, mime::JSON, mime::BYTE_STREAM])?;
146            if reponse_ty != mime::PLAIN {
147                return Ok(next.run(req).await);
148            }
149            // This is a metrics request, abort the rest of the dispatching chain and run the
150            // metrics handler.
151            let route = &req.state().clone().modules[&api].versions[&version][&route];
152            let state = &*req.state().clone().state;
153            let req = request_params(req, route.params()).await?;
154            route
155                .handle(req, state)
156                .await
157                .map_err(|err| match err {
158                    RouteError::AppSpecific(err) => err,
159                    _ => Error::from_route_error(err),
160                })
161                .map_err(|err| err.into_tide_error())
162        }
163        .boxed()
164    }
165}

167pub(crate) async fn request_params<State, Error: crate::Error>(
168    req: tide::Request<Arc<App<State, Error>>>,
169    params: &[RequestParam],
170) -> Result<RequestParams, tide::Error> {
171    RequestParams::new(req, params)
172        .await
173        .map_err(|err| Error::from_request_error(err).into_tide_error())
